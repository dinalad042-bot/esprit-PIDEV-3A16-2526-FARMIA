    function previewImage(event) {
        var file = event.target.files[0];
        if (!file) return;
        var reader = new FileReader();
        reader.onload = function(e) {
            var output  = document.getElementById('avatarPreview');
            var wrapper = document.getElementById('currentAvatarWrapper');
            if (output)  { output.src = e.target.result; output.style.display = 'block'; }
            if (wrapper) { wrapper.style.display = 'none'; }
        };
        reader.readAsDataURL(file);
    }

    function deleteFaceData() {
        fetch("/auth/face/delete", {
            method: 'POST',
            headers: { 'X-Requested-With': 'XMLHttpRequest' }
        })
        .then(r => r.json())
        .then(data => {
            if (data.success) window.location.reload();
            else alert(data.error || "Erreur lors de la suppression.");
        })
        .catch(err => alert("Erreur r├⌐seau : " + err.message));
    }

    document.addEventListener('DOMContentLoaded', function() {
        const form = document.getElementById('profileUpdateForm');
        if (!form) return;

        form.addEventListener('submit', function(e) {
            e.preventDefault();

            // Clear all errors
            document.querySelectorAll('.pf-field-error').forEach(el => el.textContent = '');
            document.querySelectorAll('.pf-input-error').forEach(el => el.classList.remove('pf-input-error'));
            const globalErrors = document.getElementById('profileErrors');
            globalErrors.style.display = 'none';
            document.getElementById('profileErrorsList').textContent = '';

            const submitBtn = document.getElementById('profileSubmitBtn');
            const original  = submitBtn.innerHTML;
            submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Traitement...';
            submitBtn.disabled  = true;

            fetch("/profile/update", {
                method: 'POST',
                headers: { 'X-Requested-With': 'XMLHttpRequest' },
                credentials: 'same-origin',
                body: new FormData(form)
            })
            .then(response => {
                if (response.status === 401) {
                    window.location.href = "/login";
                    return Promise.reject('logout');
                }
                return response.json().then(data => ({ status: response.status, body: data }));
            })
            .then(res => {
                submitBtn.innerHTML = original;
                submitBtn.disabled  = false;

                if (res.status === 200 && res.body?.success) {
                    window.location.reload();
                    return;
                }

                const errs = res.body?.errors || {};
                let hasFieldError = false;

                // errors is now a key->message dict
                if (typeof errs === 'object' && !Array.isArray(errs)) {
                    Object.entries(errs).forEach(([field, msg]) => {
                        const errDiv = document.getElementById('err-' + field);
                        const input  = document.getElementById('pf-' + field)
                                    || form.querySelector('[name="' + field + '"]');
                        if (errDiv) { errDiv.textContent = msg; hasFieldError = true; }
                        if (input)  { input.classList.add('pf-input-error'); }
                    });
                }

                if (!hasFieldError) {
                    // Fallback: show as list
                    const msgs = Array.isArray(errs) ? errs : Object.values(errs);
                    document.getElementById('profileErrorsList').textContent = msgs.join(' | ') || 'Erreur inconnue.';
                    globalErrors.style.display = 'block';
                }

                // Scroll to top of body to reveal errors
                document.querySelector('#profileModal .profile-modal-body').scrollTop = 0;
            })
            .catch(err => {
                if (err === 'logout') return;
                submitBtn.innerHTML = original;
                submitBtn.disabled  = false;
                document.getElementById('profileErrorsList').textContent = 'Erreur r├⌐seau de communication avec le serveur.';
                document.getElementById('profileErrors').style.display = 'block';
            });
        });
    });

<script>
    var faceRegModal = document.getElementById('faceRegistrationModal');
    var faceRegVideo = document.getElementById('faceRegVideo');
    var faceRegCanvas = document.getElementById('faceRegCanvas');
    var btnCaptureRegFace = document.getElementById('btnCaptureRegFace');
    var faceRegOverlay = document.getElementById('faceRegOverlay');
    var faceRegError = document.getElementById('faceRegError');
    var faceRegSuccess = document.getElementById('faceRegSuccess');
    
    var regStream = null;

    function openFaceRegistrationModal() {
        faceRegModal.style.display = 'flex';
        faceRegError.style.display = 'none';
        faceRegSuccess.style.display = 'none';
        faceRegOverlay.style.display = 'none';
        btnCaptureRegFace.disabled = false;
        
        navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" } })
            .then(function(s) {
                regStream = s;
                faceRegVideo.srcObject = regStream;
            })
            .catch(function(err) {
                faceRegError.textContent = "Impossible d'acc├⌐der ├á la webcam : " + err.message;
                faceRegError.style.display = 'block';
                btnCaptureRegFace.disabled = true;
            });
    }

    function closeFaceRegistrationModal() {
        faceRegModal.style.display = 'none';
        if (regStream) {
            regStream.getTracks().forEach(track => track.stop());
            regStream = null;
        }
    }

    // Fermer si clic au dehors
    window.addEventListener('click', function(event) {
        if (event.target == faceRegModal) {
            closeFaceRegistrationModal();
        }
    });

    if (btnCaptureRegFace) {
        btnCaptureRegFace.addEventListener('click', function() {
            if (!regStream) return;
            
            faceRegError.style.display = 'none';
            faceRegSuccess.style.display = 'none';
            faceRegOverlay.style.display = 'flex';
            btnCaptureRegFace.disabled = true;

            // Capture image
            faceRegCanvas.width = faceRegVideo.videoWidth;
            faceRegCanvas.height = faceRegVideo.videoHeight;
            const ctx = faceRegCanvas.getContext('2d');
            ctx.drawImage(faceRegVideo, 0, 0, faceRegCanvas.width, faceRegCanvas.height);
            
            const base64Image = faceRegCanvas.toDataURL('image/jpeg', 0.8);

            fetch('/auth/face/register', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'X-Requested-With': 'XMLHttpRequest'
                },
                body: JSON.stringify({ image: base64Image })
            })
            .then(response => response.json().then(data => ({status: response.status, body: data})))
            .then(res => {
                faceRegOverlay.style.display = 'none';
                if (res.status === 200 && res.body.success) {
                    faceRegSuccess.style.display = 'block';
                    setTimeout(() => {
                        closeFaceRegistrationModal();
                    }, 2000); // Close after 2s
                } else {
                    faceRegError.textContent = res.body.error || "Erreur lors de l'enregistrement de votre visage.";
                    faceRegError.style.display = 'block';
                    btnCaptureRegFace.disabled = false;
                }
            })
            .catch(err => {
                faceRegOverlay.style.display = 'none';
                faceRegError.textContent = "Erreur r├⌐seau: " + err.message;
                faceRegError.style.display = 'block';
                btnCaptureRegFace.disabled = false;
            });
        });
    }
</script>
